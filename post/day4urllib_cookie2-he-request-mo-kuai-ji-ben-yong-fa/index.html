3、代码登陆获取cookie

#coding=utf-8
import ssl #正常请求的时候需要ssl证书验证
import urllib.request
from http import cookiejar
from urllib import parse
import time
timestamp=str((time.time())*1000)
ssl._create_default_https_context = ssl._create_unverified_context #全局取消证书验证
"""
获取个人中心的页面
1、代码登陆 登陆成功 cookie(有效）
2、自动带着cookie去请求个人中心
"""
#1、代码登陆：
    #1.1登陆的网址
url="https://www.yaozh.com/login/"
    #1.2登陆的参数
loging_form_data={
    "username":"qq1403586993",
    "pwd":"zhanghaibin0.",
    "formhash":"AF8818A924",
    "backurl":"%2F%2Fwww.yaozh.com"
    }

#1、参数需要转译
login_str=parse.urlencode(loging_form_data).encode("utf-8")
print(login_str)
#发送post登陆请求
cookie_jar=cookiejar.CookieJar()
#定义有添加cook功能的处理器
cook_hanlder=urllib.request.HTTPCookieProcessor(cookie_jar)
#根据处理器生产opener
opener=urllib.request.build_opener(cook_hanlder)
#带着参数发送post请求
headers = {"User-Agent": "Mozilla/5.0 (Macintosh; Intel Mac OS X 10_14_6) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/79.0.3945.130 Safari/537.36"}
login_request=urllib.request.Request(url=url,headers=headers,data=login_str)
#如歌登陆成功，cookjar自动保存cookie
#1.2发送登陆请求
a=opener.open(login_request)
dataa=a.read().decode("utf-8")
#2.带着cooke去访问个人中心
center="https://www.yaozh.com/member/"
#2.1创建请求头
center_request=urllib.request.Request(url=center,headers=headers)

5、requests_模块基本用户

#coding=utf-8
import requests
class RequstSpider(object):
    def __init__(self):
        url='https://www.baidu.com'
        headers={
            'User-Agent':"Mozilla/5.0 (Macintosh; Intel Mac OS X 14_14_6) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/79.0.3945.130 Safari/537.36"
        }
        self.response=requests.get(url,headers=headers)

    def run(self):
        data=self.response.content
        #获取求情头
        request_headers=self.response.request.headers
        print(request_headers)
        #获取响应头
        response_headers=self.response.headers
        print(response_headers)
        #获取状态码
        code=self.response.status_code
        print(code)
        #获取请求的cookie
        request_cookie=self.response.request._cookies
        print(request_cookie)
        #获取响应的cookie
        response_cookie=self.response.cookies
        print(response_cookie)
    
if __name__=='__main__':
    RequstSpider().run()
