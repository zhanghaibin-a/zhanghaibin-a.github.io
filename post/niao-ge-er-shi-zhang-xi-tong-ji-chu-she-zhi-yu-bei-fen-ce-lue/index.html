3、nmcli设置网络并写入配置文件

[root@vms001 ~]#nmcli connection modify eno33554968 \ 
connection.autoconnect yes \
ipv4.method manual  \
ipv4.addresses 192.168.1.111/24 \ 
ipv4.gateway 192.168.1.1  \
#重启网卡
[root@vms001 ~]#nmcli connection up eno33554968
[root@vms001 ~]#nmcli connection show eno33554968
...........
IP4.ADDRESS[1]:                         192.168.1.111/24
IP4.GATEWAY:                            192.168.1.1
IP6.ADDRESS[1]:                         fe80::20c:29ff:fe41:a77b/64
自动获取 DHCP
[root@vms001 ~]# nmcli connection modify eno33554968 \ 
connection.autoconnect \
yes ipv4.method auto

时间的调整

[root@vms001 ~]# timedatectl set-time "2019-3-6 02:38"
[root@vms001 ~]# date 
Wed Mar  6 02:38:02 CST 2019
[root@vms001 ~]# timedatectl set-time
#timedatectl set-time 修改日期后，不用在修改hwclock去修改bios时间，timedatectl一口气搞定

语系设定

[root@vms001 ~]# cat /etc/locale.conf  #语系配置文件
LANG="zh_CN.UTF-8"
[root@vms001 ~]# localectl  
   System Locale: LANG=zh_CN.UTF-8 #系统语系
       VC Keymap: cn
      X11 Layout: cn
[root@vms001 ~]# localectl  #tab补全可以看到一些设置
list-keymaps    list-locales    set-keymap      set-locale      set-x11-keymap  status
[root@vms001 ~]# locale 
LANG=zh_CN.UTF-8 ##当前软件的语系
LC_CTYPE="en_US.utf8"
LC_NUMERIC="en_US.utf8"
......
LC_MEASUREMENT="en_US.utf8"
LC_IDENTIFICATION="en_US.utf8"
LC_ALL=en_US.utf8

1、修改语系
[root@vms001 ~]# localectl set-locale LANG=zh_CN.utf-8
或
[root@vms001 ~]# export LC_ALL=zh_CN.utf-8
