正则表达式re.split and re.compile

##coding=utf-8
import re
#1、拆分字符串
one='absbfsjsnogs'
#标准是s拆封
pattern = re.compile('s')
result = pattern.split(one)
print(result) #['ab', 'bf', 'j', 'nog', '']

#2、匹配中文
two = "<span>弹幕礼仪 </span>"
#python中 匹配中问 [a-z] unicode的范围* + ?
pattern = re.compile('[\u4e00-\u9fa5]')
result = pattern.findall(two)
print(result)

正则表达式re.compile实际用法

##coding=utf-8
import re
import requests
url='http://news.baidu.com/'
headers={
    "User-Agent":'Mozilla/5.0 (Macintosh; Intel Mac OS X 10_14_6) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/80.0.3987.87 Safari/537.36'
}
data=requests.get(url,headers=headers).content.decode()
#正则解析数据
#每个新闻的标题和url
#<a href="http://news.cctv.com/2020/02/20/ARTI3bDTB7a20grMxTbxfuvZ200220.shtml" target="_blank" class="a3" mon="ct=1&amp;a=1&amp;c=top&amp;pn=0">11通电话2场会见 习近平这些话展示中国担当</a>
pattern =re.compile('<a href="(.*)" target="(.*?)" mon="(.*?)">(.*)</a>',re.S)#re.S 解决.*不能匹配/N
result=pattern.findall(data)
print(result)
with open('1news.html','w') as i :
    i.write(data)

html解析库Xpath实际用法

##coding=utf-8
import requests
#安装支持解析html和xml都解析库 lxml
import json
from lxml import etree
class BtcSpider(object):
    def __init__(self):
        self.base_url='https://www.chainnode.com'
        self.headers={
    "User-Agent":'Mozilla/5.0 (Macintosh; Intel Mac OS X 10_14_6) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/80.0.3987.87 Safari/537.36'
}
    #1.发送请求
    def get_response(self):
        response=requests.get(url=self.base_url,headers=self.headers)
        #网页编码head-meta-charset
        data=response.content.decode()
        return data
    #2.解析数据
    def parse_data(self,data):
        #使用xpath解析当前页面标题和和url保存
        #1.转类型
        x_data=etree.HTML(data)
        #2.根据xpath路径解析
        #路径1.纯手写，2.借助浏览器右击粘贴xpath路径；需要修改
        title_list= x_data.xpath('//a[@class="link-dark-major font-bold bbt-block"]/text()')
        url_list=x_data.xpath('//a[@class="link-dark-major font-bold bbt-block"]/@href')
        data_list=[]
        for index,title in enumerate(title_list):
            news={}
            news['name']=title.replace(' ','').replace('\n','')
            news['url']="https://www.chainnode.com"+url_list[index]
            data_list.append(news)
            print(news)
        return data_list
    #3.保存数据
    def save_data(self,data):
        #将list--str
        data_str=json.dumps(data)
        with open('btc.json','w') as i :
            i.write(data_str)
    #4.启动
    def run(self):
        # 1.拼接完整url
        self.base_url=self.base_url+'/forum/1-10'
        print(self.base_url)
        #2.发请求
        data=self.get_response()
        #3.做解析
        list_data=self.parse_data(data)
        #4.保存
        self.save_data(list_data)

BtcSpider().run()
